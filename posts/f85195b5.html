5.标题统计

发表于2024-10-17|更新于2024-10-17
|阅读量:

题目：
代码：
#include
#include
#include
using namespace std;

int main(){
string s;
getline(cin,s); //知识点1
int length = s.size();
for(int i = 0; i < s.size(); i++){
if(s[i] - 32 == 0 || s[i] == ‘\n’){
length–;
}
}
cout<<length;
return 0;
}
官方代码：
#include<bits/stdc++.h>
using namespace std;
int main(){
int ans=0;
char c;
if(cin>>c)ans++; //cin自动去除空格换行
if(cin>>c)ans++; //cin在读不到数据时返回0
if(cin>>c)ans++; //因为题目规定标题不超过5个字符，所以可这样做
if(cin>>c)ans++;
if(cin>>c)ans++;
cout<<ans;
}

1.getline(cin,s)和 get(s)
